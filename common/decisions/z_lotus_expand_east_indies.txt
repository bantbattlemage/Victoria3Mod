net_expand_east_indies = {

	is_shown = {
		country_is_in_europe = yes
		exists = c:DEI
		c:DEI = {
			is_subject_of = ROOT
		}
		any_state = {
			owner = ROOT
			region = sr:region_indonesia
		}
	}

	possible = {
		OR = { 
			any_state = {
				owner = ROOT
				region = sr:region_indonesia
			}
			any_subject_or_below = {
				capital.region = sr:region_indonesia
				NOT = { 
					THIS = c:DEI
					is_subject_of = c:DEI
				}
			}
		}
	}

	when_taken = {
		custom_tooltip = expand_east_indies_tt
		if = {
			limit= {
				any_state = {
					owner = ROOT
					region = sr:region_indonesia
				}
			}
			every_scope_state = {
				limit = {
					owner = ROOT
					region = sr:region_indonesia
				}
				set_state_owner = c:DEI
				set_state_type = unincorporated
			}
		}
	}
	ai_chance = {
		base = 1000
	}
}

centralize_east_indies = {
	
	is_shown = {
		country_is_in_europe = yes
		exists = c:DEI
		
		c:DEI = {
			is_subject_of = ROOT
		}
	}

	possible = {

		has_technology_researched = nationalism
		is_at_war = no

		exists = c:DEI
		c:DEI = {
			is_subject_of = ROOT
			is_at_war = no
		}

		any_subject_or_below = {			

			is_player = no
			capital.state_region.region = sr:region_indonesia

			OR = {						
				NOT = { this = c:DEI }
				AND = {
					this = c:DEI
					is_subject_type = subject_type_dominion
				}
			}
		}
	}

	when_taken = {

		every_subject_or_below = {
			
			if = {
				limit = {
					is_player = no
					capital.state_region.region = sr:region_indonesia
					NOT = { this = c:DEI }
				}
			
				c:DEI = {
					annex = prev
				}			
			}
		}

		root = {

			if = {
				limit = {
					has_diplomatic_pact = {
						who = c:DEI
						type = dominion
					}
				}

				hidden_effect = {
					remove_diplomatic_pact = {
						country = c:DEI
						type = dominion
					}
				}
	
				create_diplomatic_pact = {
					country = c:DEI
					type = puppet				
				}
			}
		}
	}

	ai_chance = {
		base = 1000
	}
}

net_fix_independent_dei = {

	is_shown = {

		exists = c:NET
		exists = c:DEI
		root = c:NET
		
	}

	possible = {

		#is_at_war = no
		exists = c:DEI
		root = c:NET		

		c:DEI = {
			
			is_at_war = no
			is_player = no
			overlord = { is_player = no }
			NOT = { overlord = c:NET }
		}
	}

	when_taken = {
		
		c:DEI = {
			
			if = {
				limit = {
					has_diplomatic_pact = { 
						who = c:NET
						type = rivalry
					}
				}

				remove_diplomatic_pact = {
					country = c:NET
					type = rivalry
				}
			}

			set_relations = { 
				country = c:NET
				value = 100
			}

			create_diplomatic_pact = {
				country = c:NET
				type = dominion	
			}
		}

		c:NET = {
			set_relations = {
				country = c:DEI
				value = 100
			}
		}
	}

	ai_chance = {
		base = 1000
	}
}

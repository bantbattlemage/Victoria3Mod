gbr_fix_independent_raj = {

	is_shown = {

		exists = c:BIC
		exists = c:GBR
		root = c:GBR
		
	}

	possible = {

		#is_at_war = no
		exists = c:BIC
		root = c:GBR

		c:BIC = {
			
			#is_at_war = no
			is_player = no
			overlord = { is_player = no }
			NOT = { overlord = c:GBR }
		}
	}

	when_taken = {
		
		c:BIC = {

			if = {
				limit = {
					has_diplomatic_pact = {
						who = c:GBR
						type = rivalry
					}
				}

				remove_diplomatic_pact = {
					country = c:GBR
					type = rivalry
				}
			}

			set_relations = {
				country = c:GBR
				value = 100
			}
		}

		c:GBR = {

			set_relations = { 
				country = c:BIC
				value = 100
			}

			if = {
				limit = {
					c:BIC = { has_variable = british_raj_var }		
				}

				create_diplomatic_pact = {
					country = c:BIC
					type = puppet				
				}
			}
			else = {
				create_diplomatic_pact = {
					country = c:BIC
					type = dominion				
				}
			}
		}
	}

	ai_chance = {
		base = 1000
	}
}
